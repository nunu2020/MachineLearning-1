def get_top10_index(data_set):
    """
    :param data_set: dataframe :[df1, df2,...]
    :return: dataframe,label,[new_df1, new_df2,...] 
    """
    size_array = [df.shape[0] for df in data_set]
    vec_sum = []
    for df in data_set:
        vecs = df["GreyVec"].values
        for i,_ in enumerate(vecs):
            vecs[i] = np.array(vecs[i])
        vec_sum.append(sum(vecs))
    vec_sum = np.array(vec_sum)
    
    # 
    mean_vec = sum(vec_sum)/sum(size_array)
    top10_indexs = np.argsort(-mean_vec)[0:10]
    print("Top 10 indexes are:",top10_indexs)


    # Create New DataFrames
    
    print("Generating new data frames...")
    columns = ["X1","X2","X3","X4","X5","X6","X7","X8","X9","X10","label"]
    new_dfs = []
    for df in data_set:
        new_df_values = []
        vecs = df["GreyVec"].values
        labels = df["label"].values
        for i,_ in enumerate(vecs):
            vecs[i] = np.array(vecs[i])
            vecs[i] = vecs[i][top10_indexs]
            vecs[i] = list(vecs[i])
            vecs[i].append(labels[i])
            new_df_values.append(vecs[i])
        new_df_values = np.array(new_df_values)
        new_df = pd.DataFrame(data=new_df_values, columns=columns)
        new_dfs.append(new_df)
    print("finished successfully!")
    return new_dfs, top10_indexs

#sigmod
def sigmoid(inX):
    return 1.0/(1+np.exp(-inX))

#
def gradAscent(dataMatIn, classLabels):
    dataMatrix =  np.mat(dataMatIn)             #
    labelMat = np.mat(classLabels).transpose() #labelMat
    m,n = np.shape(dataMatrix)
    
    alpha=0.001
    
    #
    maxCycles =5000
    weights = np.ones((n,1))
    #for
    for k in range(maxCycles):  
        h = sigmoid(dataMatrix*weights)
        error = (labelMat - h)  
        #print('',dataMatrix.transpose()* error)
        weights = weights+alpha*dataMatrix.transpose()* error 

    return weights

#FNRFPRprecisionrecall
def calsta(clas,l):
    fnr=np.zeros(l+1)
    fpr=np.zeros(l+1)
    pre=np.zeros(l+1)
    rec=np.zeros(l+1)
    for i in range(l+1):
        test=np.zeros(l)
        #
        TP=0
        FP=0
        TN=0
        FN=0
        for j in range(l-i):
            test[j]=1
        for j in range(l):
            if clas[j]==test[j] and test[j]==1:
                TP=TP+1
            elif clas[j]==test[j] and test[j]==0:
                TN=TN+1
            elif clas[j]!=test[j] and test[j]==1:
                FP=FP+1
            elif clas[j]!=test[j] and test[j]==0:
                FN=FN+1
        if(TP+FP!=0):
            p=TP/(TP+FP)
        else :
            p=1
        if(TP+FN!=0):
            fn=FN/(TP+FN)
        else:
            fn=1
        if((FP+TN)!=0):
            f=FP/(FP+TN)
        else :
            f=1
        if(TP+FN!=0):
            r=TP/(TP+FN)
        else:
            r=1
        fnr[i]=fn
        fpr[i]=f
        pre[i]=p
        rec[i]=r
    
    return fnr,fpr,pre,rec

def AUC_cal(fp,re):
    result=0
    for i in range(len(fp)-1):
        result+=(fp[i]-fp[i+1])*(re[i+1]+re[i])
    result=result/2
    return result

def bubble_sort(truth,model):
    for i in range(len(truth)-1):
        for j in range(len(truth)-1-i):
            if(model[j+1]>model[j]):
                model[j+1],model[j]=model[j],model[j+1]
                truth[j+1],truth[j]=truth[j],truth[j+1]
    return truth,model

def threshold(fp,re):
    m=(1-fp)*re
    th=np.argmax(m)
    return th

def matrix(clas,th):
    test=np.zeros(len(clas))
    TP=0
    FP=0
    TN=0
    FN=0
    for i in range(th):
        test[i]=1
    for j in range(len(clas)):
        if clas[j]==test[j] and test[j]==1:
            TP=TP+1
        elif clas[j]==test[j] and test[j]==0:

            FP=FP+1
        elif clas[j]!=test[j] and test[j]==0:
            FN=FN+1
    return pd.DataFrame([[TP,FN],[FP,TN]],index=["P","N"],columns=["P","N"])

from sklearn.model_selection import KFold
import random

def Fold_5(X,Y):#5'real','SO_p','SO','sag_p','sag','liblinear_p','liblinear'7
    seq = np.arange(0,len(X))
    random.shuffle(seq)
    
    X = X[seq]
    Y = Y[seq]

    kf = KFold(n_splits=5)
    tr_index = []
    te_index = []
    result = []

    for train_index, test_index in kf.split(X):
        #train_indextest_index
        tr_index.append(train_index)
        te_index.append(test_index)
    
        train_X = X[train_index]
        test_X = X[test_index]
        train_Y = Y[train_index]
        test_Y = Y[test_index]
    
        Y_real,prob1,Y1 = Logistic(train_X,test_X,train_Y,test_Y,'SO',3,6)
        Y_real,prob2,Y2 = Logistic(train_X,test_X,train_Y,test_Y,'sag',3,6)
        Y_real,prob3,Y3 = Logistic(train_X,test_X,train_Y,test_Y,'liblinear',3,6)
    
        result.append(np.c_[Y_real,prob1,Y1,prob2,Y2,prob3,Y3])
    
    return result

    def __init__(self):

        self.eig_vecs = None
    
    def mean_vectors(self,X, y):
        """
        :param X ,y train_set
        :return: mean vectors for each class
        """
        np.set_printoptions(precision=4)
        mean_vectors = []
        for c in [self.c1, self.c2]:
            mean_vectors.append(np.mean(X[y==c], axis=0))
            
        print("mean vecotrs generated!")
        
        return mean_vectors
    
    def with_in_class_scatter(self,X,y,mean_vectors):
        """
        :param X ,y train_set mean vectors for each class
        :return: within class scatter
        """
        S_W = np.zeros((X.shape[1],X.shape[1]))
        for c, m in zip([self.c1, self.c2], mean_vectors):
            class_sc_mat = np.zeros((X.shape[1],X.shape[1]))
            for row in X[y==c]:
                row, m = row.reshape(X.shape[1],1), m.reshape(X.shape[1],1)
                class_sc_mat += (row-m).dot((row-m).T)
            S_W += class_sc_mat
        self.w = np.linalg.inv(S_W).dot(mean_vectors[1]-mean_vectors[0])
        
        print("S_w generated!")
        
        return S_W
    
    def between_class_scatter(self,X,y,mean_vectors):
        overall_mean = np.mean(X, axis=0)
        S_B = np.zeros((X.shape[1],X.shape[1]))
        for i, m in zip([self.c1, self.c2], mean_vectors):
            n = X[y==i,:].shape[0]
            m = m.reshape(X.shape[1],1)
            overall_mean = overall_mean.reshape(X.shape[1],1)
            S_B += n*(m-overall_mean).dot((m-overall_mean).T)
        return S_B
    
    def eigenvalue(self,dx, S_W, S_B):
        eig_vals, eig_vecs = np.linalg.eig(np.linalg.inv(S_W).dot(S_B))
        for i in range(len(eig_vals)):
            eigvec_sc = eig_vecs[:,i].reshape(dx,1)
        return eig_vals, eig_vecs
    
    def explained_var(self):
        pairs = [(np.abs(self.eig_vals[i]), self.eig_vecs[:,i]) for i in range(len(self.eig_vals))]
        pairs = sorted(pairs, key=lambda x: x[0], reverse=True)
        for pair in pairs:
            print(pair[0])
        eig_val_sums = sum(self.eig_vals)
        print('Explained Variance')
        for i, pair in enumerate(pairs):
            print('Eigenvector {}: {}'.format(i, (pair[0]/eig_val_sums).real))
    
    def get_divide(self):
        """
        return : based on meanctors to get the mean division
        """
        mu0_w = np.dot(self.mean_vecs[0], self.w.T)
        mu1_w = np.dot(self.mean_vecs[1], self.w.T)
        divide = 0.5*(mu0_w+mu1_w)
        return divide
    
    def get_best_w0(self, w, X_train, y_train):
        print("Getting best w0 to classify...")

class myQDA():
    def __init__(self):
        self.p1 = 0.5
        self.p2 = 0.5
        self.mu1 = None
        self.mu2 = None
        
        #qda
        self.cov1 = None
        self.cov2 = None
        self.c1 = None
        self.c2=None
        self.d = 2
    
    def mean_vectors(self,X, y):
        
        np.set_printoptions(precision=4)
        mean_vectors = []
        for c in np.unique(y):
            mean_vectors.append(np.mean(X[y==c], axis=0))
            
        return mean_vectors
    
    def get_cov_matrixs(self, X,y):
        covs = []
        for c in np.unique(y):
            covs.append(np.cov(X[y==c].T))
            
        return covs
    
    def get_class_name(self, y):
        c = []
        for e in np.unique(y):
            c.append(e)
            
        return c
    
    def get_prior_prob(self,X,y):
        n1 = X[y==self.c1].shape[0]
        n2 = X[y==self.c2].shape[0]
        
        return n1/(n1+n2), n2/(n1+n2)
    
    def fit(self, X, y):
        #
        [self.mu1, self.mu2] = self.mean_vectors(X, y)
        [self.cov1,self.cov2] = self.get_cov_matrixs(X, y)
        [self.c1, self.c2] = self.get_class_name(y)
        self.p1, self.p2 = self.get_prior_prob(X,y)

    def predict(self, X, y):
        y_pred = []
        covi_1 = np.linalg.inv(self.cov1)
        covi_2 = np.linalg.inv(self.cov2)
        det_1 = np.linalg.det(self.cov1)
        det_2 = np.linalg.det(self.cov2)
        for x in X:
            r1_sq = 0.5*(x-self.mu1).T.dot(covi_1).dot(x-self.mu1)
            r2_sq = 0.5*(x-self.mu2).T.dot(covi_2).dot(x-self.mu2)
            if r1_sq < r2_sq + np.log(self.p1/self.p2) + 0.5*np.log(det_2/det_1):
                y_pred.append(self.c1)
            else:
                y_pred.append(self.c2)

        res = sum(y==y_pred)
        
        return res/len(y)

class ecoc():
    def __init__(self):
        self.mean_vecs = []
        self.positive_labels = []
        self.negative_labels = []
        self.n_index = []
        self.p_index = []
        self.covi = None
    
    def get_mean_vecs(self, X, y_label, pos_labels, neg_labels):
        
        p_index = []
        n_index = []
        for y in y_label:
            if y in pos_labels:
                p_index.append(True)
                n_index.append(False)
            else:
                p_index.append(False)
                n_index.append(True)
        self.p_index = p_index
        self.n_index = n_index
        self.mean_vecs.append(np.mean(X[p_index], axis=0))
        self.mean_vecs.append(np.mean(X[n_index], axis=0))
        
    def get_Sw(self, X):
        
        class_sc_mat1 = np.zeros((10,10))
        class_sc_mat2 = np.zeros((10,10))
        S_w = np.zeros((10,10))
        
        for row in X[self.p_index]:
            row = row.reshape(10,1)
            mv = self.mean_vecs[0].reshape(10,1)
            class_sc_mat1 += (row-mv).dot((row-mv).T)
        S_w += class_sc_mat1
        
        for row in X[self.n_index]:
            row = row.reshape(10,1)
            mv = self.mean_vecs[1].reshape(10,1)
            class_sc_mat1 += (row-mv).dot((row-mv).T)
        S_w += class_sc_mat2
        self.covi = np.linalg.inv(S_w/(X.shape[0]-2))
    
    def fit(self, X, y,pos_labels, neg_labels):
        self.get_mean_vecs(X,y,pos_labels, neg_labels)
        self.get_Sw(X)
    
    def predict(self, X_test, y_test):
        pred = []
        for x in X_test:
            dp = np.sqrt((x-self.mean_vecs[0]).T.dot(self.covi).dot(x-self.mean_vecs[0]))
            dn = np.sqrt((x-self.mean_vecs[1]).T.dot(self.covi).dot(x-self.mean_vecs[1]))
            if dp<dn:
                pred.append(1)
            else:
                pred.append(-1)
        
        return pred
